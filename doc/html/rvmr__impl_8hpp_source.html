62 void RVMR<KernelType>::Train(const arma::mat& data,
63  const arma::rowvec& responses){
64 
65  arma::mat phi;
66  arma::rowvec t;
67 
68  // Manage the kernel
69  if (this->ardRegression == false)
70  {
71  // We must keep the original training data for future predictions.
72  this->phi = data;
73  applyKernel(data, data, phi);
74 
75  //Preprocess the data. Center and normalize.
76  preprocess_data(phi,
77  responses,
78  this->fitIntercept,
79  this->normalize,
80  phi,
81  t,
82  this->data_offset,
83  this->data_scale,
84  this->responses_offset);
85  }
86 
87  else
88  {
89  //Preprocess the data. Center and normalize.
90  preprocess_data(data,
91  responses,
92  this->fitIntercept,
93  this->normalize,
94  phi,
95  t,
96  this->data_offset,
97  this->data_scale,
98  this->responses_offset);
99  }
100 
101  unsigned short p = phi.n_rows, n = phi.n_cols;
102  // Initialize the hyperparameters and
103  // begin with an infinitely broad prior.
104  this->alpha_threshold = 1e4;
105  this->alpha = arma::ones<arma::rowvec>(p)*1e-6;
106  this->beta = 1 / (arma::var(t) * 0.1);
107 
108  // Loop variables.
109  double tol = 1e-5;
110  double L = 1.0;
111  double crit = 1.0;
112  unsigned short nIterMax = 50;
113  unsigned short i = 0;
114  unsigned short ind_act;
115 
116  arma::rowvec gammai = arma::zeros<arma::rowvec>(p);
117  arma::mat matA;
118  arma::rowvec temp(n);
119  arma::mat subPhi;
120  // Initiaze a vector of all the indices from the first
121  // to the last point.
122  arma::uvec allCols(n);
123  for (size_t i=0; i < n; i++) {allCols[i] = i;}
124 
125  while ((crit > tol) && (i < nIterMax))
126  {
127  crit = -L;
128  this->activeSet = find(this->alpha < this->alpha_threshold);
129  // Prune out the inactive basis function. This procedure speeds up
130  // the algorithm.
131  subPhi = phi.submat(this->activeSet, allCols);
132 
133  // Compute the posterior statistics.
134  matA = diagmat(this->alpha.elem(this->activeSet));
135  this->matCovariance = inv(matA
136  + (subPhi
137  * subPhi.t())
138  * this->beta);
139 
140  this->omega = (this->matCovariance
141  * subPhi
142  * t.t())
143  * this->beta;
144 
145  // Update the alpha_i.
146  for (size_t k=0; k<this->activeSet.size(); k++)
147  {
148  ind_act = this->activeSet[k];
149  gammai[ind_act] = 1-this->matCovariance(k,k) *
150  this->alpha[ind_act];
151 
152  this->alpha[ind_act] = gammai[ind_act]
153  / (this->omega[k] * this->omega[k]);
154  }
155 
156  // Update beta.
157  temp = t - this->omega.t() * subPhi;
158  this->beta = (n - sum(gammai.elem(this->activeSet))) / dot(temp, temp);
159 
160  // Comptute the stopping criterion.
161  L = norm(this->omega);
162  crit = abs(crit + L) / L;
163  i++;
164  }
165 }
